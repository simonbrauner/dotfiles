* EXWM

A full-featured [[https://github.com/ch11ng/exwm][tiling X window manager]] for Emacs. Example configuration [[https://github.com/ch11ng/exwm/wiki/Configuration-Example][here]].

#+begin_src emacs-lisp
(use-package exwm
  :config (exwm-enable))
#+end_src

** Keys

Executing programs.

#+begin_src emacs-lisp
(defun sb/start-process (command)
  (start-process-shell-command command nil command))

(defun sb/execute-application (command)
  (interactive (list (read-shell-command "$ ")))
  (sb/start-process command))

(defun sb/open-browser ()
  (interactive)
  (sb/start-process "firefox"))

(defun sb/toggle-volume ()
  (interactive)
  (sb/start-process "amixer set Master toggle"))

(defun sb/lower-volume ()
  (interactive)
  (sb/start-process "amixer set Master 5%-"))

(defun sb/raise-volume ()
  (interactive)
  (sb/start-process "amixer set Master 5%+"))

(defun sb/mic-mute ()
  (interactive)
  (sb/start-process "amixer set Capture nocap"))

(defun sb/brightness-down ()
  (interactive)
  (sb/start-process "brightnessctl --min-value=10 s 15%-"))

(defun sb/brightness-up ()
  (interactive)
  (sb/start-process "brightnessctl --min-value=10 s 15%+"))
#+end_src

Global keybindings.

#+begin_src emacs-lisp
(setq exwm-input-global-keys
      `(;; super prefix
        (,(kbd "s-r") . exwm-reset)
        (,(kbd "s-p") . sb/execute-application)
        (,(kbd "s-w") . sb/open-browser)
        (,(kbd "s-t") . multi-vterm)
        ;; leptop keys
        (,(kbd "<XF86AudioMute>") . sb/toggle-volume)
        (,(kbd "<XF86AudioLowerVolume>") . sb/lower-volume)
        (,(kbd "<XF86AudioRaiseVolume>") . sb/raise-volume)
        (,(kbd "<XF86AudioMicMute>") . sb/mic-mute)
        (,(kbd "<XF86MonBrightnessDown>") . sb/brightness-down)
        (,(kbd "<XF86MonBrightnessUp>") . sb/brightness-up)))
#+end_src

** Hooks

Naming buffers reasonably.

#+begin_src emacs-lisp
(add-hook 'exwm-update-class-hook
          (lambda ()
            (unless (or (string-prefix-p "sun-awt-X11-" exwm-instance-name)
                        (string= "gimp" exwm-instance-name))
              (exwm-workspace-rename-buffer exwm-class-name))))

(add-hook 'exwm-update-title-hook
          (lambda ()
            (when (or (not exwm-instance-name)
                      (string-prefix-p "sun-awt-X11-" exwm-instance-name)
                      (string= "gimp" exwm-instance-name))
              (exwm-workspace-rename-buffer exwm-title))))
#+end_src

* Polybar

A fast and easy-to-use [[https://github.com/polybar/polybar][tool for creating status bars]].

** Face

Designed to fit in the right part of Emacs' echo area.

#+begin_src conf
[bar/poly]
override-redirect = true
bottom = true
fixed-center = false

width = 20%
offset-x = 80%
height = 15

foreground = #ffffff
background = #282a36

font-0 = Fira Code:pixelsize=11:semibold;1

padding = 1
module-margin = 2
modules-right = pulseaudio bluetooth battery date
#+end_src

** Modules

Volume and bluetooth.

#+begin_src conf
[module/pulseaudio]
type = internal/pulseaudio

label-volume = "%percentage%%  "
label-muted = %percentage%% m

interval = 5

[module/bluetooth]
type = custom/script

exec = sh ~/.config/polybar/bluetooth

interval = 5
#+end_src

Print =b= if bluetooth headphones are connected.

#+begin_src shell :tangle ~/.config/polybar/bluetooth
if [ $(bluetoothctl info | wc -l) -ne 1 ]; then
    echo "b"
else
    echo " "
fi
#+end_src
