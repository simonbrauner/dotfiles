#+title: Programs

* Xmonad

** Configuration

*** Imports

#+begin_src haskell :tangle ~/.xmonad/xmonad.hs :mkdirp yes
import XMonad
import XMonad.Actions.SpawnOn
#+end_src

*** Basics

Default programs, super key as leader key and no red border.

#+begin_src haskell :tangle ~/.xmonad/xmonad.hs :mkdirp yes
myEditor   = "emacs"
myBrowser  = "firefox"
myTerminal = "kitty"

myBorderWidth = 0

myModMask = mod4Mask
#+end_src

*** Workspaces

Nine workspaces, first three of them have applications on startup.

#+begin_src haskell :tangle ~/.xmonad/xmonad.hs :mkdirp yes
myWorkspaces = map show [1..9]

myManageHook = manageSpawn <+> manageHook def

myStartupHook = do
  spawnOn "1" myEditor
  spawnOn "2" myBrowser
  spawnOn "3" myTerminal
#+end_src

*** Main

#+begin_src haskell :tangle ~/.xmonad/xmonad.hs :mkdirp yes
main = xmonad defaults

defaults = def
  { terminal    = myTerminal
  , borderWidth = myBorderWidth
  , modMask     = myModMask
  , workspaces  = myWorkspaces
  , manageHook  = myManageHook
  , startupHook = myStartupHook
  }
#+end_src

** Executing with startx

Starting the windows manager.

#+begin_src shell :tangle ~/.xinitrc
exec xmonad
#+end_src

** Recompiling after update

Hook for pacman.

#+begin_src conf :tangle /sudo::/etc/pacman.d/hooks/xmonad.hook :mkdirp yes
[Trigger]
Operation = Upgrade
Type = Package
Target = xmonad

[Action]
Description = Recompiling xmonad...
When = PostTransaction
Exec = /usr/bin/sudo -u sb /usr/bin/xmonad --recompile
#+end_src
