* Xmonad

** Configuration
:properties:
:header-args+: :tangle ~/.xmonad/xmonad.hs
:end:

*** Imports

#+begin_src haskell
import XMonad
import XMonad.Actions.SpawnOn
import XMonad.Layout.NoBorders
#+end_src

*** Basics

Default programs, super key as leader key.

#+begin_src haskell
myEditor   = "emacs"
myBrowser  = "firefox"
myTerminal = "kitty"

myModMask = mod4Mask
#+end_src

*** Layout

Borders - colors, size, visible only with multiple windows. =Tall= layout.

#+begin_src haskell
myNormalBorderColor  = "#19194d"
myFocusedBorderColor = "#0033cc"

myBorderWidth = 3

myLayoutHook = smartBorders $ Tall 1 (3/100) (1/2)
#+end_src

*** Workspaces

Nine workspaces, first three of them have applications on startup.

#+begin_src haskell
myWorkspaces = map show [1..9]

myManageHook = manageSpawn <+> manageHook def

myStartupHook = do
  spawnOn "1" myEditor
  spawnOn "2" myBrowser
  spawnOn "3" myTerminal
#+end_src

*** Main

#+begin_src haskell
main = xmonad defaults

defaults = def
  { terminal           = myTerminal
  , modMask            = myModMask
  , normalBorderColor  = myNormalBorderColor
  , focusedBorderColor = myFocusedBorderColor
  , borderWidth        = myBorderWidth
  , layoutHook         = myLayoutHook
  , workspaces         = myWorkspaces
  , manageHook         = myManageHook
  , startupHook        = myStartupHook
  }
#+end_src

** Recompiling after update

Hook for pacman.

#+begin_src conf :tangle /sudo::/etc/pacman.d/hooks/xmonad.hook
[Trigger]
Operation = Upgrade
Type = Package
Target = xmonad

[Action]
Description = Recompiling xmonad...
When = PostTransaction
Exec = /usr/bin/sudo -u sb /usr/bin/xmonad --recompile
#+end_src
